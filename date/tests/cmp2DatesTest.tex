\newcommand{\testIfBeforeDate}[6]{
  \cmpTwoDates{#1}{#2}{#3}{#4}{#5}{#6}

  {#1}.{#2}(month).{#3} is
  %%package "ifthen"
  \ifthenelse{\leftDateCmpRightDate=1}%%if
    {\textbf{before} }%%then
    {}%%else
  %%package "ifthen"
  \ifthenelse{\leftDateCmpRightDate=2}%%if
    {\textbf{after} }%%then
    {}%%else
  %%package "ifthen"
  \ifthenelse{\leftDateCmpRightDate=0}%%if
    {\textbf{same as} }%%then
    {}%%else
   {#4}.{#5}(month).{#6}
}
