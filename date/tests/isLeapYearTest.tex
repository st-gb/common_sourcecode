\newcommand{
%%Prefix with author's TU Berlin matriculation number to avoid name clashes.
%%TUBlnThSiTenNiFi=TU Berlin THree SIx TEN NIne FIve
  \TUBlnThSiTenNiFiTestIfLeapYear}[
%%Parameter #1:year to test if it is a leap year
%%Parameter #2:1=expected result:leap year, 0=expected result:no leap
  2]
{
  \TUBlnThSiTenNiFiSetVarVal{\TUBlnThSiTenNiFiTestIfLeapYearYear}{#1}
  \TUBlnThSiTenNiFiSetVarVal{\TUBlnThSiTenNiFiTestIfLeapYearExpctdRslt}{#2}
  \TUBlnThSiTenNiFiIsLeapYear{\TUBlnThSiTenNiFiTestIfLeapYearYear}
 
  \TUBlnThSiTenNiFiTestIfLeapYearYear
    \ifthenelse{\value{TUBlnThSiTenNiFiLeapYear}=1}%%"if" branch
      { \textbf{is} }%%"then" branch
      { is \textbf{not} }%%"else" branch
    a leap year:\newline
%%Make comment to prevent a new paragraph on an empty new line.
%%
  year(\TUBlnThSiTenNiFiTestIfLeapYearYear) mod 4:
   \theTUBlnThSiTenNiFiIsLeapYearYearModFour
  \ifthenelse{
  	\theTUBlnThSiTenNiFiIsLeapYearYearModFour=0}%%"if" branch
    { year mod 4 = 0 $\implies$ potential leap year}%%"then" branch
    { $\implies$ \textbf{no} leap year}%%"else" branch
  \newline
  year(\TUBlnThSiTenNiFiTestIfLeapYearYear) mod 100:
   \theTUBlnThSiTenNiFiIsLeapYearYearModOneHoundred
  \ifthenelse{
  	\theTUBlnThSiTenNiFiIsLeapYearYearModOneHoundred=0}%%"if" branch
    { year mod 100 = 0 $\implies$ potentially no leap year}%%"then" branch
    {}%%"else" branch
  \newline
  year(\TUBlnThSiTenNiFiTestIfLeapYearYear) mod 400:
   \theTUBlnThSiTenNiFiIsLeapYearYearModFourHoundred
  \ifthenelse{
  	\theTUBlnThSiTenNiFiIsLeapYearYearModFourHoundred=0}%%"if" branch
    { year mod 400 = 0 $\implies$ leap year}%%"then" branch
    {}%%"else" branch
  \newline
  \ifthenelse{%%"if" branch
    \TUBlnThSiTenNiFiTestIfLeapYearExpctdRslt=\value{TUBlnThSiTenNiFiLeapYear}
   }
   {%%"then" branch
    \color{green}
    result fits expected result
    \color{black}
   }
   {%%"else" branch
    \color{green}
    result does not fit expected result
    \color{black}
   }
}
